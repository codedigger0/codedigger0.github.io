css的引入方式

 /* 内嵌式 */
  <style>
        /* css的注释 */
        /* css写在stlye标签中 */
        /* 选择器:{css属性} */
        /* 选择器查找标签 */
      
        p{
            /*文字颜色变红  */
            color: red;

            /* 字变大  px:像素*/
            font-size: 30px;

            /* 背景颜色 */
            background-color: yellowgreen;

            /* width height */
            width: 400px; height: 400px;

        }
    </style>
</head>
<body>
    <p>这是一个p标签</p>
</body>

<head>
    <!-- 内嵌式：css写在style标签中  通常写在head标签中-->
    <!-- 外联式：css写在单独的.css文件中 通过link标签引入到html中-->
    <!-- 行内式：css写在stlye属性中 -->
    <!-- 关系：样式表 -->
    <!-- css样式用link标签引入 -->
    <link rel="stylesheet" href="./my.css">
</head>
<body>
    <p>这是一个p标签</p>
    <!-- 行内式css -->
    <div style=" color: aqua; font-size: 20px;">这是div标签</div>

标签选择器

<style>
        /* 选择器{} */
        /* 标签选择器 就是 以标签命名的选择器 */
        /* 标签选择器 选中的这个标签生效css样式 */
        p{
            color: red;
        }
    </style>
</head>
<body>
    <p>你是一个小可爱</p>
    <p>你是一个小可爱</p>
    <p>你是一个小可爱</p>
    <p>你是一个小可爱</p>
</body>

类选择器

<style>
    /* <!-- 类选择器 结构 .类名{ css属性名：属性值；} --> */
    /* <!-- 所有标签都有class属性 class 属性的属性值为类名 --> */
    /* <!-- 不能以数字和中划线开头 --> */
    /* <!-- 类名可以重复，一个选择器可以同时选中多个标签 --> */
    /* <!-- 类：定义 和 使用才能生效 --> */
    /* <!-- 针对性的控制某一个标签 --> */
        .red{
            color: #ff5e00;
        }

        .size{
            font-size: 30px;
        }
    </style>
</head>
<body>
    

    <p>dddddd</p>

    <!-- 一个标签可以同时有多个类名类名之间以空格隔开 -->
    <p class="red size" >dddddd</p>
    <p>dddddd</p>

    <div class="red">想把这个文字变红</div>
</body>

id选择器

 <style>
        /* 结构 ： #id属性值{css属性名：属性值} */
        /* 所有标签都有id属性值 */
        /* id属性值在页面是唯一的 不可重复的 */
        /* 一个标签只能有一个id属性值 */
        /* 一个id选择器只能选中一个标签 */
        
        /* 定义id选择器 */
        #blue{
            color: blue;
        }
    </style>
</head>
<body>
    <div id="blue">这是div标签哈哈哈哈</div>
</body>

通配符选择器

<style>
        /* 结构 ： *{css属性名：属性值；} */
        /* 去除标签默认的margin和padding  清除默认的内外间距*/
        *{
            color: red;
            margin: 0;
            padding: 0;
        }
    </style>
</head>
