    <!-- 相对路径同级目录 -->

音频标签

    <!-- 音频标签audio -->
    <!--src 音频的路径  -->
    <!-- controls 显示播放的控件 -->
    <!-- autoplay 自动播放  部分浏览器不支持自动播放-->
    <!-- loop 循环播放 -->
    <!-- 音频标签支持 MP3 Wav Ogg这三种格式 -->
    <audio src="添加音频的路径" controls autoplay muted loop>

视频标签

    <!-- 视频标签video -->
    <!-- src 视频的路径 -->
    <!-- controls 显示播放的控件 -->
    <!-- autoplay 自动播放谷歌浏览器需配合muted实现静音播放 -->
    <!-- loop 循环播放 -->
    <!-- 视频支持MP4 webm ogg 尽量使用mp4 -->
    <video src="添加视频标签" width="200" controls autoplay muted loop></video>
