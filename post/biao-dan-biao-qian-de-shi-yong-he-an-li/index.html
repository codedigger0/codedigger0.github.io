input标签的使用和属性

<!-- input标签可以通过type属性值的不同，展示不同效果 -->
    <!-- text 文本框，输入单行文本 -->
    <!-- password 密码框，用于输入密码 -->
    <!-- radio 单选框，用于多选一 -->
    <!-- checkbox 多选框，用于多选多 -->
    <!-- file 文件选择，用于上传文件 -->
    <!-- submit 提交按钮，用于提交 -->
    <!-- reset 重置按钮，用于重置 -->
    <!-- button 普通按钮 默认无功能，配合js添加功能 -->

    文本：<input type="text">
    <br>
    <br>
    <!-- 书写内容都会显示成*** -->
    密码框：<input type="password">
    <br>
    <br>
    单选框：<input type="radio">
    <br>
    <br>
    多选框：<input type="checkbox">
    <br>
    <br>
    上传文件：<input type="file">

#占位符

    <!-- placeholder 占位符，提示用户输入内容的文本 -->
    <input type="text" placeholder="请输入用户名" >

    <input type="password" placeholder="请输入密码">

#单选框

    <!-- name 分组，有相同name属性值的单选框为一组，一组中同时只能有一个被选中 -->
    <!-- checked 默认选中 -->
    性别<input type="radio" name="sex" checked>男
        <input type="radio" name="sex">女

    <br>
    <br>
    <!-- 多选框默认选中 -->
    <input type="checkbox" checked>

#上传文件

    <!-- multiple 上传文件多文件选择 -->
    <input type="file" multiple>

#按钮input

        <!-- subimt 提交按钮，点击之后提交数据给后端服务器 -->
        <!-- reset 重置按钮， 点击之后恢复表单的默认值 -->
        <!-- button 普通按钮，默认无功能 -->
        <!-- form 作为数据父标签 -->
        <!-- action="数据提交的地址" -->
         <form action="">
            用户名：<input type="text" placeholder="请输入用户名">
        <br>
        <br>
        密码：<input type="password " placeholder="请输入密码">
        <br>
        <br>
        <input type="submit" value="免费注册">
        <input type="reset">
        <!-- value 给按钮添加文字显示 -->
        <!-- 属性  xx=“xxxx” -->
        <input type="button" value="普通按钮">

         </form>

#按钮button

        <!-- subimt 提交按钮，点击之后提交数据给后端服务器 -->
        <!-- reset 重置按钮， 点击之后恢复表单的默认值 -->
        <!-- button 普通按钮，默认无功能 -->
        <!-- button 相当于自定义按钮 -->
        <button>我是按钮</button>

        <button type="submit">提交按钮</button>
        <button type="reset">重置按钮</button>
        <button type="button">普通按钮 没有功能</button>

#下拉菜单

    <!-- select 在网页中提供多个选择项的下拉菜单表单控件 -->
    <!-- select 下拉菜单的 整体-->
    <!-- option 下拉菜单的每一项 -->
    <!-- selected 下拉菜单的默认选中 -->
    <!-- select与option 是父子系关系 -->
    <select >
        <option selected>北京</option>
        <option >上海</option>
        <option >广州</option>
        <option >深圳</option>
        
    </select>

#文本域

    <!-- textarea 在网页中提供可输入多行文本的表单控件 -->
    <!-- cols 规定文本域内可见宽度 -->
    <!-- rows 规定文本域内可见行数 -->
    <!-- 右下角可拖拽改变大小 -->
    <textarea  cols="30" rows="10"></textarea>

#lable标签

    <!-- 使用lable标签 把内容/文本包裹起来 -->
    <!-- 在表单上添加id属性 -->
    <!-- 在lable标签的for属性设置对应的id属性值 -->
    性别:    <input type="radio" name="sex" id="nan"><label for="nan">男</label>

            <input type="radio" name="sex" id="nv"><label for="nv">女</label>

    <!-- 直接使用lable标签把内容和表单标签一起包裹起来 -->
    <!-- 把lable标签饿for属性删除即可 -->

    性别：  

        <label ><input type="radio" name="1">男</label>
        <label ><input type="radio" name="1">女</label>
        

        <!-- 不能够这样书写 这样把两个单选框都选中了。一个lable包裹一个单选框 -->
        <label >
            <input type="radio" name="1">男
            <input type="radio" name="1">女
        </label>

#没有语义的标签

    <!-- div标签：一行只显示一个 独占一行 -->
    <!-- span标签：一行可以显示多个 -->
    <div>这是div标签</div>
    <div>这是div标签</div>
    <span>这是span标签</span>
    <span>这是span标签</span>

#有语义的标签

<!--和div显示特点一致 比div多了语义 -->

    <header>网页头部</header>
    <nav>网页导航</nav>
    <footer>网页底部</footer>
    <aside>侧边栏</aside>
    <section>网页区块</section>
    <article>网页的文章</article>

#表格综合案例

<caption><h1>优秀学生信息表</h1></caption>
    <table border="2" width="300" height="300" >
        <thead>
            <tr>
                <th>年级</th>
                <th>姓名</th>
                <th>学号</th>
                <th>班级</th>
            </tr>
        </thead>

        <tbody>
            <tr>
                <!-- rowspan 垂直合并 -->
                <td rowspan="2">高三</td>
                <td>黑蛋</td>
                <td>99</td>
                <td>一年级二班</td>
            </tr>
    
            <tr>
                
                <td>狗蛋</td>
                <td>89</td>
                <td>一年级三班</td>
            </tr>
        </tbody>
        
        <tfoot>
            <tr>
                <td>评语</td>
                <!-- colspan 水平合并 -->
                <td colspan="3">你们都很优秀</td>
                
            </tr>
        </tfoot>
 </table>
